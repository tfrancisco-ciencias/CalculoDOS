\section*{Integrabilidad de funciones continuas} \item Considera la funci\'on $f(x)=mx+b$, $x\in \mathbb{R}$, con $m\not=0$. Dada $\varepsilon$ define $\delta=\delta(\varepsilon) := \frac{\varepsilon}{|a|}$. Demuestra que si $|x-y|<\delta$ entonces $ |f(x)-f(y)|<\varepsilon $. \item Considera la funci\'on $f(x)=\frac{1}{x}$, $x\ne 0$. Fija $x_0\ne 0$. Dada $\varepsilon$, define $\delta=\delta(x_0, \varepsilon):= \min\{ \varepsilon |x_0|^2/2, \frac{|x_0|}{2} \}$. Demuestra que si $|x-x_0|<\delta$ entonces $ |f(x_0)-f(x)|<\varepsilon $. \begin{definicion} Sea $D\subset \mathbb{R}$ y $f:D \to \mathbb{R}$ una funci\'on. Decimos que $f$ es uniformemente continua en $D$ si, para toda $\varepsilon>0$, existe $\delta>0$, que s\'olo depende de $\varepsilon$ tal que, para cualesquiera $x,y\in D$ con $|x-y|<\delta$ se satisface que $$ |f(x)-f(y)|<\varepsilon. $$ Nota: la $\delta$ debe de ser independiente de los puntos $x$ e $y$. \end{definicion} \item Define $f:[a,b] \to \mathbb{R}$ por $f(x)=x^2$. Usando la definici\'on de continuidad uniforme, demuestra que $f$ es uniformemente continua en $[a,b]$. Sugerencia: $x^2-y^2=(x+y)(x-y)$. \item\label{Ejer:ContinuidadUniformeEnCompactos} Demuestra el siguiente teorema. \begin{teorema}[Continuidad Uuniforme de funciones continuas en intervalos cerrados y acotados] Sea $f:[a,b]\to \mathbb{R}$ una funci\'on continua en $[a,b]$. Para todo $\varepsilon >0$, existe $\delta>0$, que s\'olo depende de $\varepsilon$, tal que: si $x,y\in [a,b]$ satisfacen $|x-y|<\delta$ entonces $|f(x)-f(y)| < \varepsilon$. \end{teorema} Sugerencia: procede por contradicci\'on de la siguiente forma: \begin{enumerate} \item sup\'on que existe $\varepsilon_0>0$ tal que para toda $\delta>0$ existen puntos, que dependen de $\delta$, $x, y \in [a,b]$ con $|x - y |<\delta$ y $|f(x) - f(y)| \geq \varepsilon_0$; \item tomando $\delta=1/n$ construye sucesiones $(x_n)_{n=1}^\infty$, $(y_n)_{n=1}^\infty$ que satisfacen: $|x_n - y_n| < \frac{1}{n}$ y $|f(x_n)-f(y_n)| \geq \varepsilon_0$, para toda $n$; \item usa el Teorema de Bolzano (\ref{Ejer:TeoremaBolzanoWeirestrass}) para asegurar que existen subsucesiones $(x_{n_k})_{k=1}^\infty$, $(y_{n_k})_{k=1}^\infty$ y $x_0\in [a,b]$ con $\lim_{k \to \infty}x_{n_k}=x_0=\lim_{k\to \infty} y_{n_k}$; \item finalmente llega a una contradicci\'on usando la continuidad de $f$ en $x_0$. \end{enumerate} \begin{definicion}[Oscilaci\'on o di\'ametro] Sea $I \ne \emptyset$ un intervalo y $f:I\to \mathbb{R}$. La oscilaci\'on o di\'ametro de $f$ en $I$ se define como $$ \omega_f(I)=\sup_{x_1,x_2\in I}\{ |f(x_1)-f(x_2)|\} $$ Nota: si el conjunto $\{ |f(x_1)- f(x_2) : x_1,x_2\in I|\}$ no es acotado superiormente definimos $\omega_f(I)=\infty$. \end{definicion} \item Demuestra que $\omega_f(J)\leq \omega_f(I)$, si $J \subseteq I$. \item Calcula las siguientes oscilaciones. \begin{enumerate} \item Si $f(x)=mx+b$, $I=[c,d]$, $\omega_f (I)=|m|(d-c) $ \item Si $f(x)=1/x$, $I=(0,1)$, $\omega_f (I)=\infty$. \end{enumerate} \item Sea $f:[a,b]\to \mathbb{R}$ continua en $[a,b]$. Fija $\varepsilon >0$ y sea $\delta >0$, la de la continuidad uniforme de $f$ para $\varepsilon$ (la del ejercicio \ref{Ejer:ContinuidadUniformeEnCompactos}). Demuestra que: $$ \omega_f(I)\leq \varepsilon, $$ para todo $I \subseteq [a,b]$, sub-intervalo de longitud menor a $\delta$. \item\label{Ejer:IntegrabilidadFuncionesContinuas} Sea $f:[a,b]\to \mathbb{R}$ continua en $[a,b]$. Este ejercicio demuestra que $f$ es integrable. \begin{enumerate} \item Denota $\Delta_n=\frac{b-a}{n}$ y $P_n=\{ a+ i\Delta_n \}_{i=1}^n$ la partici\'on homogenea de $[a,b]$ de longitud $n$. Define funciones escalonadas \begin{eqnarray*} s_n&=&\sum_{i=1}^n f(x_{i *})\chi_{[a+(i-1)\Delta_n,a+i\Delta_n)}\\ t_n&=&\sum_{i=1}^n f(x_i^*)\chi_{[a+(i-1)\Delta_n,a+i\Delta_n)} \end{eqnarray*} donde, para cada $i$, $f(x_{i*})=\min\{f(x): x\in [x_{i-1},x_i]\}$ y $f(x_i^*)=\max\{f(x): x\in [x_{i-1}, x_i] \}$. Demuestra que $s_n \leq f \leq t_n$. %\begin{eqnarray*} %%\int_a^b s_n(x) \leq \int_a^b f(x)dx \leq \int_a^bt(x)dx %\end{eqnarray*} \item Denota $$ \Sigma_n = \frac{b-a}{n}\sum_{i=1}^n f(a+i\Delta_n). $$ Demuestra que: $$ \int_a^b s_n(x) \leq \Sigma_n \leq \int_a^bt(x)dx. $$ \item Demuestra que: $$ \int_a^b t_n(x)dx-\int_a^bs_n(x)dx \leq \frac{b-a}{n} \sum_{i=1}^n \omega_f([a+(i-1)\Delta_n, a+i\Delta_n]). $$ \item Usando la continuidad de $f$, el ejercicio \ref{Ejer:ContinuidadUniformeEnCompactos} y los incisos anteriores demuestra que $f$ es integrable en $[a,b]$ y $$ \int_a^b f(x)dx=\lim_n \frac{b-a}{n} \sum_{i=0}^{n-1} f\left(a+ \frac{i}{n}(b-a) \right) $$ \end{enumerate} \item\label{Ejer:AproximacionSumasRiemann} Sea $f:[a,b]\to \mathbb{R}$ continua en $[a,b]$. Sea $P=\{ x_i\}_{i=0}^n$ la partici\'on homogenea con longitud $n$. Para cada $i$, tomemos puntos $r_{i}$ en el sub-intervalo $[x_{i-1}, x_i]$. \begin{enumerate} \item Demuestra $$ \int_a^bf(x)dx=\sum_{i=1}^n f(r_{i,n})(x_i-x_{i-1})+Error, $$ donde $$ |Error| \leq \frac{(b-a)}{n} \sum_{i=1}^n \omega_f ([x_{i-1},x_i]) $$ Sugerencia: toma $s_n, t_n$ como en el ejercicio \ref{Ejer:IntegrabilidadFuncionesContinuas}. Define $$ \rho_n=\sum_{i=1}^n r_{i} \chi_{[x_{i-1}, x_i)}. $$ Demuestra que $s_n \leq \rho_n \leq t_n$, $\int_a^b \rho_n =\sum_{i=1}^n f(r_{i,n})(x_i-x_{i-1})$ y concluye que $$ \left| \int_a^bf(x)dx -\sum_{i=1}^n f(r_{i})(x_i-x_{i-1}) \right| \leq \int_a^b t_n -\int_a^b s_n $$ \item Concluye que $$ \lim_{n\to \infty} \sum_{i=1}^n f(r_{i})(x_i-x_{i-1})= \int_a^b f(x)dx. $$ \end{enumerate} \begin{definicion} Sea $f:[a,b]\to \mathbb{R}$ una funci\'on. Una suma de Riemann de $f$ es una expresi\'on de la forma $$ \sum_{i=1}^n f(r_i)[x_{i}-x_{i-1}] $$ donde $P=\{x_i\}_{i=0}^n$ una partici\'on de $[a,b]$ y $r_i \in [x_{i-1}, x_i]$, $i=1,\dots, n$, puntos fijos y arbitrarios. Entonces, por el ejercicio \ref{Ejer:AproximacionSumasRiemann}, para funciones continuas, sus sumas de Riemann con particiones homogeneas se aproximan a la integral de la funci\'on. \end{definicion} \item Expresa los siguiente l\'imites como integrales. No es necesario evaluar la integral. \begin{enumerate} \item $\lim_{n\to \infty} \frac{1}{n} \sum_{i=1}^n e^{(3i)/n}$. \item $\lim_{n\to \infty} \frac{\pi}{n} \sum_{i=1}^n \sen\left(\pi+\frac{\pi i}{n} \right)$. \item $\lim_{n\to \infty} \frac{1}{n} \sum_{i=1}^n \sqrt{2+\frac{i}{n}}$. \item $\lim_{n\to \infty} \frac{1}{n} \sum_{i=1}^n \frac{1}{3+(i/n)^2}$ \end{enumerate} \item Fija $a,b \in \mathbb{R}$ con $0 0 $ para alg\'un $x_0\in [a,b]$ demuestra que: $$ \int_a^b f(x)dx >0 $$ \item\label{Ejer:IntegralCeroDePositivas} Sea $f:[a,b]\to \mathbb{R}$ continua en $[a,b]$ con $f(x)\geq 0 $ para todo $x\in [a,b]$. Si $\int_a^bf(x)dx=0$ demuestra que $f(x)=0$ para toda $x\in [a,b]$. Sugerencia: usa el ejercicio \ref{Ejer:LemaIntegralCeroDePositivas}. \item Da un ejemplo de una funci\'on continua $f:[a,b]\to \mathbb{R}$ tal que $\int_a^b f(x)dx>0$ pero que no se cumpla $f(x) \geq 0$ para todo $x\in [a,b]$. \item Este ejercicio muestra la relaci\'on entre la oscilaci\'on de una funci\'on y la continuidad. Considera una funci\'on $f:[a,b]\to \mathbb{R}$ y $x_0$ un punto fijo en el intervalo. Prueba que $$ \lim_{\delta \to 0}\omega f([x_0-\delta, x_0+\delta])=0 $$ si y s\'olo si $f$ es continua en $x_0$. \item\label{Ejer:TeoremaValorMedioIntegrales} Demuestra el Teorema del valor medio para para integrales. Sea $f:[a,b]\to \mathbb{R}$ una funci\'on continua. Existe $\xi \in [a,b]$ con $$ \int_a^b f(x)dx=f(\xi)(b-a) $$ Sugerencia: considera la desigualdad $$ \min_{x\in [a,b]}\{f(x)\} \leq f(x) \leq \max_{x\in [a,b]}\{f(x)\} $$ integra y aplica el Teorema del Valor Intermedio. \item\label{Ejer:TeoremaValorPonderadoIntegrales} Demuestra el Teorema del valor medio ponderado para integrales. Supongamos que $f,g:[a,b]\to \mathbb{R}$ son funciones continuas y que $g$ nunca cambia de signo en el intervalo $[a,b]$. Entonces existe $\xi \in [a,b]$ tal que $$ \int_a^bf(x)g(x)=f(\xi)\int_a^bg(x)dx $$ Sugerencia: si $g(x) \geq 0$ para todo $x\in [a,b]$, considera la desigualdad $$ \min_{x\in [a,b]}\{f(x)\} g(x) \leq f(x)g(x) \leq \max_{x\in [a,b]}\{f(x)\} g(x) $$ integra y aplica el Teorema del Valor Intermedio a $f$.
